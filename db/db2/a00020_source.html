210 
211  //Get location, vel, and angle
212  double fire_v;
213  double angle;
214  double radius = (double)global::equations::kTimeSizeRatio * HoldTime;
215 
216  fire_v = -1 * sqrt( pow(mouseC.x - mouseO.x, 2) + pow(mouseC.y - mouseO.y, 2) );
217  fire_v *= (double) global::equations::kVelocityScalar;
218 
219  //If the mouse if pointing straight up or straight down make the angle 90
220  //Otherwise calculate the angle with atan.
221  if (mouseC.x == mouseO.x) {
222  angle = (M_PI / 2.0) * ( (mouseC.y > mouseO.y) ? -1.0 : 1.0 );
223  } else {
224  angle = (double) -1.0 * atan( (double)(mouseC.y - mouseO.y) /
225  (double)(mouseC.x - mouseO.x) );
226  angle += (double)( (mouseC.x < mouseO.x) ? M_PI : 0.0 );
227  } //end if x = x
228 
229  //mod mouse start, once again because the top left is 0,0 to SDL

283 void cannonballs::doCollide(uint numA, uint numB) {
293 
294  dblXY Avel, Bvel;
295  PP Aprops, Bprops;
296 
297  Avel = balls[numA].getVelocity();
298  Bvel = balls[numB].getVelocity();
299  Aprops = balls[numA].getPhysicalProps();
300  Bprops = balls[numB].getPhysicalProps();
301 
302 #ifdef DEFINED_USE_R2_VEL_MODDER
303  //This part here has no actual basis on real life,
304  //it is just my attempt at preventing the cannonballs from sticking together
305  LOC CenterA, CenterB, DeltaCenters;
306  double VelModder;
307  CenterA = balls[numA].getplace();
308  CenterB = balls[numB].getplace();
309 
310  DeltaCenters.x = abs(CenterA.x - CenterB.x);
311  DeltaCenters.y = abs(CenterA.y - CenterB.y);
312  //Since it is r^2 and the sqrt of this give us r, we just drop the sqrt part to save time
313  VelModder = sqrt(pow((double)DeltaCenters.x,2) + pow((double)DeltaCenters.y,2));
314  VelModder = sqrt(200.0) / VelModder; //200 is the highest value possible
315  if (VelModder < 1.0) {VelModder = 1.0;}
316 
317  Avel.x *= (double) VelModder;
318  Avel.y *= (double) VelModder;
319  Bvel.x *= (double) VelModder;
320  Bvel.y *= (double) VelModder;
321  //End of non real stuff
322 #endif
323 
324  double Aangle, Bangle, ContactAngle;
325  double Atotal_v, Btotal_v;
326 
327  dblXY TotalAMomentum, TotalBMomentum;
328 
329  if ( global::physics::collisionmethod != CollidePerfectInelastic ) {
330  //The equations for Perfect Inelastic is much simpler than if not
331  //so I am handling those a bit differently to speed up that method.
332 
333  //Equations used can be found and explained here:
334  //https://en.wikipedia.org/wiki/Elastic_collision
335  Atotal_v = sqrt( pow(Avel.x,2) + pow(Avel.y,2) );
336  Btotal_v = sqrt( pow(Bvel.x,2) + pow(Bvel.y,2) );
337 
338  //get the angle for both A and B
339  if (Avel.x != 0.0) { Aangle = atan(Avel.y/Avel.x); }
340  else { Aangle = (Avel.y >= 0.0) ? M_PI / 2.0 : -M_PI / 2.0; }
341 
342  if (Bvel.x != 0.0) { Bangle = atan(Bvel.y/Bvel.x); }
343  else { Bangle = (Bvel.y >= 0.0) ? M_PI / 2.0 : -M_PI / 2.0; }
344 
345  //Adjust the angle to be the right one.
346  //Since atan will only yield a number between -PI/2 and PI/2 we
347  //have to adjust it if xvel is negative.
348  Aangle += Avel.x < 0.0 ? M_PI : 0;
349  Bangle += Bvel.x < 0.0 ? M_PI : 0;
350  //The contact angle has to be the difference between the two angles but
351  //since sometimes one or the other is negative, we'll use abs to ensure the right number
352  ContactAngle = abs ( abs(Aangle) - abs(Bangle) );
353 
354  TotalAMomentum.x = Atotal_v * cos(Aangle - ContactAngle) *
355  (Aprops.mass - Bprops.mass);
356  TotalAMomentum.x += 2.0 * Bprops.mass * Btotal_v *
357  cos(Bangle - ContactAngle);
358  TotalAMomentum.x /= (Aprops.mass + Bprops.mass);
359  //x and y formulas are the same until this point
360  TotalAMomentum.y = TotalAMomentum.x;
361  TotalAMomentum.x *= cos(ContactAngle);
362  TotalAMomentum.y *= sin(ContactAngle);
363  TotalAMomentum.x += Atotal_v * sin(Aangle - ContactAngle) *
364  cos(ContactAngle + (M_PI/2.0) );
365  TotalAMomentum.y += Atotal_v * sin(Aangle - ContactAngle) *
366  sin(ContactAngle + (M_PI/2.0) );
367 
368  //Now do B
369  TotalBMomentum.x = Btotal_v * cos(Bangle - ContactAngle) *
370  (Bprops.mass - Aprops.mass);
371  TotalBMomentum.x += 2.0 * Aprops.mass * Atotal_v *
372  cos(Aangle - ContactAngle);
373  TotalBMomentum.x /= (Aprops.mass + Bprops.mass);
374  //x and y formulas are the same until this point
375  TotalBMomentum.y = TotalBMomentum.x;
376  TotalBMomentum.x *= cos(ContactAngle);
377  TotalBMomentum.y *= sin(ContactAngle);
378  TotalBMomentum.x += Btotal_v * sin(Bangle - ContactAngle) *
379  cos(ContactAngle + (M_PI/2.0) );
380  TotalBMomentum.y += Btotal_v * sin(Bangle - ContactAngle) *
381  sin(ContactAngle + (M_PI/2.0) );
382  } else {
383  TotalAMomentum.x = Aprops.mass * Avel.x + Bprops.mass * Bvel.x;
384  TotalAMomentum.y = Aprops.mass * Avel.y + Bprops.mass * Bvel.y;
385  } //end if Perfect Inelastic or not
386 
387  switch (global::physics::collisionmethod) {
388  //figure out what to do based on the collision method
389  case CollidePerfectInelastic:
390  //The balls collide and stick together
391  //Get the new mass.
392  Aprops.mass += Bprops.mass;
393  //Now we have to calculate the new radius, volume, and area so that drag will reflect the new size
394  //we are of course assuming that density stays the same.
395  Aprops.volume = (Aprops.mass / (double)global::physics::kBallDensity);
396  //cbrt = cube root
397  Aprops.radius = cbrt( (double) (3.0*Aprops.volume) / (double) (4.0*M_PI) );
398  Aprops.area = (double) (2.0 * M_PI * pow(Aprops.radius, 2) );
399  //Now calculate the new velocity
400  Avel.x = TotalAMomentum.x / Aprops.mass;
401  Avel.y = TotalAMomentum.y / Aprops.mass;
402  //now "kill" cannonball B and update ball A
403  balls[numB].blnstarted_ = false;
404  balls[numA].setPhysicalProps(Aprops);
405  balls[numA].setVelocity(Avel);
406  break;
407  case CollideInelastic:
408  //uses the same equations as below but some energy is lost.
409  TotalAMomentum.x *= (double)global::physics::kCoefficientRestitution;
410  TotalAMomentum.y *= (double)global::physics::kCoefficientRestitution;
411  TotalBMomentum.x *= (double)global::physics::kCoefficientRestitution;
412  TotalBMomentum.y *= (double)global::physics::kCoefficientRestitution;
413 
414  case CollideElastic:
415  //The balls collide and bounce away from each other
416 
417  //All of the heavy lifting is handled above.
418  balls[numA].setVelocity(TotalAMomentum);
419  balls[numB].setVelocity(TotalBMomentum);
420  break;
421  default: //the catch all and CollideNone
422  //Nothing Happens!
423  break;
424  } //end switch collide method
425 } //end do Collide
