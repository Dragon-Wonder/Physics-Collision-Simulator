287  CenterB = Cannonballs::Balls[numB].getplace();

290  DeltaCenters.y = abs(CenterA.y - CenterB.y);
291  //Since it is r^2 and the sqrt of this give us r, we just drop the sqrt part to save time
292  VelModder = sqrt(pow((double)DeltaCenters.x,2) + pow((double)DeltaCenters.y,2));
293  VelModder = sqrt(200.0) / VelModder; //200 is the highest value possible
294  if (VelModder < 1.0) {VelModder = 1.0;}

300  //End of non real stuff
301 #endif

304  double Atotal_v, Btotal_v;
305 
306  dblXY TotalAMomentum, TotalBMomentum;

308  if ( Global::Physics::CollisionMethod != CollidePerfectInelastic ) {
309  //The equations for Perfect Inelastic is much simpler than if not
310  //so I am handling those a bit differently to speed up that method.
311 
312  //Equations used can be found and explained here:
313  //https://en.wikipedia.org/wiki/Elastic_collision
314  Atotal_v = sqrt( pow(Avel.x,2) + pow(Avel.y,2) );
315  Btotal_v = sqrt( pow(Bvel.x,2) + pow(Bvel.y,2) );
316

319  else {
320  Aangle = (Avel.y >= 0.0) ? PI / 2.0 : -PI / 2.0;
321  }
322 
323  if (Bvel.x != 0.0) {Bangle = atan(Bvel.y/Bvel.x);}
324  else {
325  Bangle = (Bvel.y >= 0.0) ? PI / 2.0 : -PI / 2.0;
326  }
327 
328  //Adjust the angle to be the right one.
329  //Since atan will only yield a number between -PI/2 and PI/2 we
330  //have to adjust it if xvel is negative.
331  Aangle += Avel.x < 0.0 ? PI : 0;
332  Bangle += Bvel.x < 0.0 ? PI : 0;
333  //The contact angle has to be the difference between the two angles but
334  //since sometimes one or the other is negative, we'll use abs to ensure the right number
335  ContactAngle = abs ( abs(Aangle) - abs(Bangle) );
336 
337  TotalAMomentum.x = Atotal_v * cos(Aangle - ContactAngle) * (Aprops.mass - Bprops.mass);
338  TotalAMomentum.x += 2.0 * Bprops.mass * Btotal_v * cos(Bangle - ContactAngle);
339  TotalAMomentum.x /= (Aprops.mass + Bprops.mass);
340  //x and y formulas are the same until this point
341  TotalAMomentum.y = TotalAMomentum.x;
342  TotalAMomentum.x *= cos(ContactAngle);
343  TotalAMomentum.y *= sin(ContactAngle);
344  TotalAMomentum.x += Atotal_v * sin(Aangle - ContactAngle) * cos(ContactAngle + (PI/2.0) );
345  TotalAMomentum.y += Atotal_v * cos(Aangle - ContactAngle) * cos(ContactAngle + (PI/2.0) );
346 
347  //Now do B
348  TotalBMomentum.x = Btotal_v * cos(Bangle - ContactAngle) * (Bprops.mass - Aprops.mass);
349  TotalBMomentum.x += 2.0 * Aprops.mass * Atotal_v * cos(Aangle - ContactAngle);
350  TotalBMomentum.x /= (Aprops.mass + Bprops.mass);
351  //x and y formulas are the same until this point
352  TotalBMomentum.y = TotalBMomentum.x;
353  TotalBMomentum.x *= cos(ContactAngle);
354  TotalBMomentum.y *= sin(ContactAngle);
355  TotalBMomentum.x += Btotal_v * sin(Bangle - ContactAngle) * cos(ContactAngle + (PI/2.0) );
356  TotalBMomentum.y += Btotal_v * cos(Bangle - ContactAngle) * cos(ContactAngle + (PI/2.0) );
357  } else {
358  TotalAMomentum.x = Aprops.mass * Avel.x + Bprops.mass * Bvel.x;
359  TotalAMomentum.y = Aprops.mass * Avel.y + Bprops.mass * Bvel.y;
360  } //end if Perfect Inelastic or not
361 
362  switch (Global::Physics::CollisionMethod) { //figure out what to do based on the collision method
363  case CollidePerfectInelastic:
364  //The balls collide and stick together
365  //Get the new mass.
366  Aprops.mass += Bprops.mass;
367  //Now we have to calculate the new radius, volume, and area so that drag will reflect the new size
368  //we are of course assuming that density stays the same.
369  Aprops.volume = (Aprops.mass / (double)Global::Physics::uBallDensity);
370  //cbrt = cube root
371  Aprops.radius = cbrt( (double) (3.0*Aprops.volume) / (double) (4.0*PI) );
372  Aprops.area = (double) (2.0 * PI * pow(Aprops.radius, 2) );
373  //Now calculate the new velocity
374  Avel.x = TotalAMomentum.x / Aprops.mass;
375  Avel.y = TotalAMomentum.y / Aprops.mass;
376  //now "kill" cannonball B and update ball A
377  Cannonballs::Balls[numB].blnstarted = false;
378  Cannonballs::Balls[numA].setPhysicalProps(Aprops);
379  Cannonballs::Balls[numA].setVelocity(Avel);
380  break;
381  case CollideInelastic:
382  //uses the same equations as below but some energy is lost.
383  TotalAMomentum.x *= (double)Global::Physics::fCoefficientRestitution;
384  TotalAMomentum.y *= (double)Global::Physics::fCoefficientRestitution;
385  TotalBMomentum.x *= (double)Global::Physics::fCoefficientRestitution;
386  TotalBMomentum.y *= (double)Global::Physics::fCoefficientRestitution;
387 
388  case CollideElastic:
389  //The balls collide and bounce away from each other
390 
391  //All of the heavy lifting is handled above.
392  Cannonballs::Balls[numA].setVelocity(TotalAMomentum);
393  Cannonballs::Balls[numB].setVelocity(TotalBMomentum);
394  break;
395  default: //the catch all and CollideNone
396  //Nothing Happens!
397  break;
398  } //end switch collide method
399 } //end do Collide
