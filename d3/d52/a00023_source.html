233

236  SDL_QueryTexture(screen::screenatt.pixel, NULL, NULL, &dst.w, &dst.h);

239  pow(Current.y - Old.y, 2) ) );
240  if (Current.x == Old.x ) {
241  dst.x = Current.x;
242  for (uint i = 0; i < length; ++i) {
243  dst.y = i + (Current.y > Old.y ? Old.y : Current.y);
244  SDL_RenderCopy(screen::screenatt.ren, screen::screenatt.pixel, NULL, &dst);
245  } //end for length
246  } else {
247  slope = ((double)Current.y - (double)Old.y);
248  slope /=((double)Current.x - (double)Old.x);
249  uint startpoint = (Old.x < Current.x) ? Old.x : Current.x;
250  uint endpoint = (Old.x < Current.x) ? Current.x : Old.x;
251  double incamount = (double)(endpoint - startpoint) / length;
252  for (double i = startpoint; i < endpoint; i += incamount) {
